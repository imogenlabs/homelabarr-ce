## §1 — Findings

### H-1 (carry-forward) — No CI gate on the pentest harness

**WRONG:** PRs to main can land code that breaks any of the 15 atomic tests. Nothing runs the harness in CI. The whole point of R10 was to convert one-shot audits into continuous regression coverage; that loop isn't closed yet.

**RIGHT:** Add `.github/workflows/pentest.yml`:

```yaml
name: Pentest harness
on:
  pull_request:
    branches: [main, dev]
  push:
    branches: [main]
  schedule:
    - cron: '0 6 * * *'   # daily at 06:00 UTC against prod
  workflow_dispatch:
    inputs:
      target:
        description: 'ART_TARGET URL'
        required: false
        default: 'https://ce-demo.homelabarr.com'

permissions:
  contents: read

concurrency:
  group: pentest-${{ github.ref }}
  cancel-in-progress: true

jobs:
  external-set:
    name: ATT&CK external (class A1)
    runs-on: ubuntu-latest
    timeout-minutes: 15
    steps:
      - uses: actions/checkout@<PIN-SHA>
      - name: Install jq curl openssl
        run: sudo apt-get update -qq && sudo apt-get install -y --no-install-recommends jq curl openssl
      - name: Run external harness
        env:
          ART_TARGET: ${{ github.event.inputs.target || 'https://ce-demo.homelabarr.com' }}
          ART_LOG_DIR: ${{ runner.temp }}/art-logs
        run: |
          mkdir -p "$ART_LOG_DIR"
          chmod +x pentest/harness/run.sh pentest/atomics/*/test.sh
          pentest/harness/run.sh --class A1
      - name: Upload harness logs
        if: always()
        uses: actions/upload-artifact@<PIN-SHA>
        with:
          name: art-logs-external
          path: ${{ runner.temp }}/art-logs

  internal-set:
    name: ATT&CK container-internal (class A4)
    runs-on: ubuntu-latest
    timeout-minutes: 15
    steps:
      - uses: actions/checkout@<PIN-SHA>
      - name: Build backend image
        run: docker compose build backend
      - name: Start backend
        run: |
          docker compose up -d backend
          for i in $(seq 1 30); do
            curl -sf http://localhost:3001/api/health && break || sleep 2
          done
      - name: Run internal harness
        run: |
          docker compose exec -T backend sh -c '
            cd /app && chmod +x pentest/harness/run.sh pentest/atomics/*/test.sh
            ART_TARGET=http://localhost:3001 ART_LOG_DIR=/tmp/art-logs pentest/harness/run.sh --class A4
          '
      - name: Collect internal logs
        if: always()
        run: docker compose cp backend:/tmp/art-logs ./art-logs-internal || true
      - name: Upload internal logs
        if: always()
        uses: actions/upload-artifact@<PIN-SHA>
        with:
          name: art-logs-internal
          path: art-logs-internal
```

Use the same SHA-pinned action versions you already use in `security-audit.yml` (R5 §audit). Branch protection on main must require `Pentest harness / external-set` and `Pentest harness / internal-set` to pass before merge.

### H-2 (carry-forward) — mitre_tid tagging middleware missing

**WRONG:** Audit log (R6) still emits free-text `action` field with no MITRE classification. Detection content has to grep on strings. The R10 spec required `mitre_tid` + `mitre_tactic` on every blocked event.

**RIGHT:** `server/middleware/attackTag.js`:

```js
// Tags request-failure events with MITRE ATT&CK technique/tactic IDs.
// Mounted AFTER auth middleware so it sees req.auth state.
// Mounted BEFORE the audit emit() call so it can decorate event meta.

const SIGNATURES = [
  // pattern, technique, tactic
  { match: r => r.path === '/auth/login' && r.body && r.rateLimited,        tid: 'T1110.001', tactic: 'TA0006' }, // brute force
  { match: r => r.path === '/auth/login' && r.status === 401,                tid: 'T1078',     tactic: 'TA0001' }, // valid accounts probe
  { match: r => /^\/\.(env|git|aws|ssh)/.test(r.path),                       tid: 'T1595.002', tactic: 'TA0043' }, // vuln scan
  { match: r => /wp-login|phpmyadmin|admin\.php|xmlrpc\.php/.test(r.path),  tid: 'T1595.002', tactic: 'TA0043' },
  { match: r => r.path.startsWith('/auth') && r.jwt_invalid,                  tid: 'T1606.001', tactic: 'TA0006' }, // forge token
  { match: r => r.path.includes('..') || /%2e%2e/i.test(r.originalPath),      tid: 'T1083',     tactic: 'TA0007' }, // path traversal / discovery
  { match: r => r.method !== 'GET' && r.status === 403 && !r.csrfOk,          tid: 'T1190',     tactic: 'TA0001' },
];

function attackTag(req, res, next) {
  res.on('finish', () => {
    try {
      const ctx = {
        path: req.path,
        originalPath: req.originalUrl,
        method: req.method,
        status: res.statusCode,
        body: req.body,
        rateLimited: res.statusCode === 429,
        jwt_invalid: res.locals?.jwtError === true,
        csrfOk: res.locals?.csrfOk === true,
      };
      for (const sig of SIGNATURES) {
        if (sig.match(ctx)) {
          req.audit = req.audit || {};
          req.audit.mitre_tid = sig.tid;
          req.audit.mitre_tactic = sig.tactic;
          break;
        }
      }
    } catch (_) { /* never throw from middleware */ }
  });
  next();
}

module.exports = { attackTag };
```

Wire in `server/index.js` after auth, before routes:

```js
const { attackTag } = require('./middleware/attackTag');
app.use(attackTag);
```

Update `server/audit.js` `emit()` to pull `mitre_tid` and `mitre_tactic` from `req.audit` and merge into the event payload BEFORE the hash-chain is computed (so the tag is integrity-protected, not advisory).

### H-3 (carry-forward) — Honey endpoints missing, /wp-login.php returns 200

**WRONG:** Vite SPA catch-all serves index.html for any non-API path. `/wp-login.php` returns 200 with the React shell — a scanner cannot distinguish that from a real WordPress install attempting to render. `/.env` and `/.git/config` are 403-blocked at the edge so the backend never sees the request and no audit event fires. No T1595.002 IOC is captured anywhere.

**RIGHT:** `server/routes/honey.js` mounted BEFORE the SPA catch-all:

```js
// Honey endpoints — always 404 to client, always audit event to log.
// Mount BEFORE static/SPA so the SPA catch-all never wins.

const HONEY_PATHS = [
  // dotfiles
  '/.env', '/.env.local', '/.env.production',
  '/.git/config', '/.git/HEAD',
  '/.aws/credentials', '/.ssh/id_rsa',
  // CMS scanners
  '/wp-login.php', '/wp-admin/', '/xmlrpc.php',
  '/phpmyadmin/', '/phpMyAdmin/',
  // legacy admin probes
  '/admin.php', '/administrator/', '/manager/html',
  // shell upload probes
  '/shell.php', '/cmd.jsp', '/c99.php',
  // cloud metadata SSRF probes (server-side IOC)
  '/latest/meta-data/', '/computeMetadata/v1/',
];

function mountHoney(app, audit) {
  const handler = (req, res) => {
    try {
      audit.emit({
        ts: Date.now(),
        actor: req.auth?.sub || 'anon',
        action: 'honey.hit',
        result: 'blocked',
        mitre_tid: 'T1595.002',
        mitre_tactic: 'TA0043',
        meta: {
          path: req.originalUrl,
          ip: req.ip,
          ua: (req.get('user-agent') || '').slice(0, 200),
          method: req.method,
        },
      });
    } catch (_) { /* never throw */ }
    res.status(404).type('text/plain').send('Not Found');
  };
  HONEY_PATHS.forEach(p => app.all(p, handler));
  // also catch the /api/admin/legacy-debug spec'd in R10
  app.all('/api/admin/legacy-debug', handler);
}

module.exports = { mountHoney };
```

Wire in `server/index.js`:

```js
const { mountHoney } = require('./routes/honey');
// IMPORTANT: mount BEFORE express.static and BEFORE the SPA fallback
mountHoney(app, audit);
// ... then static / SPA routes
```

Also: at the edge (nginx or Traefik), REMOVE the existing 403 deny on `/.env` and `/.git/config` so the backend can see and tag the request. The 404 from the backend is fine — the goal is the audit event, not the response body.
